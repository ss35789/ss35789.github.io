기존에 있던 정렬 알고리즘들을 필요시마다 구현할 필요없이 STL을 이용하여 짧게 사용할 수 있다.


```c++
#include <iostream>
#include <algorithm>

using namespace std;

bool compare(int a,int b){
  return a>b;
}
class Student{
   public :
    int score;
    string name;
    Student(int score,string name){
        this->score=score;
        this->name=name;
        
    }
    
    bool operator < (Student &student){
        return this->score<student.score;
        
    }//다른 학생과 비교를 할 때 내 점수가 더 낫다면 낮은 점수부터 오도록 정렬
    
    
};

int main(void){
  int array[10]={2,3,5,7,8,6,9,1,4,10};
    Student students[4]={
        Student(23,"가"),
        Student(35,"나"),
        Student(34,"라"),
        Student(26,"더")
        
        
    };
  
  sort(students,students+4);
  
  for(int i=0;i<4;i++){
    cout<< students[i].name <<' ';
  }


}
```
