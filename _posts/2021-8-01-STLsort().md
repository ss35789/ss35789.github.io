#6


기존에 있던 정렬 알고리즘들을 필요시마다 구현할 필요없이 STL을 이용하여 짧게 사용할 수 있다.


```c++
#include <iostream>
#include <algorithm>

using namespace std;

bool compare(int a,int b){
  return a>b;
}


int main(void){
  int array[10]={2,3,5,7,8,6,9,1,4,10};
  sort(array,array+10,compare);
  
  for(int i=0;i<10;i++){
    cout<< array[i] <<' ';
  }


}









```
